package main

import (
	"fmt"
	"log"
	"os"
	"strings"
)

func main() {
	if len(os.Args) < 2 {
		log.Fatal("Please provide a file path as the first argument")
	}

	filePath := os.Args[1]

	lines := strings.Split(filePath, "\n")

	for _, line := range lines {
		if strings.Contains(line, "de") {
			fmt.Println(line)
		}
	}
}
